\makeatletter
\@namedef{drama*}{%
    \@drversetrue
    \if@poemscol
        \begin{poem}
        \setcounter{verselinenumber}{\value{storelineno}}
        \setcounter{printlineindex}{\value{storeprintlineindex}}
    \else
        \begin{verse}
    \fi
    \ifx\poemlines\@undefined\else
        \setcounter{poemline}{\value{storelineno}}
    \fi
    \list{}{%
        \labelwidth\speakswidth
        \itemindent\speaksindent
        \itemsep\speechskip
        \parsep\Dparsep
        \labelsep\Dlabelsep
        \let\makelabel\speakslabel}
}
\@namedef{enddrama*}{%
    \endlist
    \ifx\poemlines\@undefined\else
        \setcounter{storelineno}{\value{poemline}}
    \fi
    \if@poemscol
        \end{poem}
        \setcounter{storelineno}{\value{verselinenumber}}
        \setcounter{storeprintlineindex}{\value{printlineindex}}
    \else
        \end{verse}
    \fi
    \@drversefalse}

\def\@character#1#2{%
  \@namedef{#2}{{\namefont #1}\xspace}

  }

\begin{document}

\Character[Subtle, The Alchemist]{Subtle}{subtle}
\Character[Face, The Housekeeper]{Face}{face}
\Character[Dol Common, Their Colleague]{Dol}{dol}
\settowidth{\speakswidth}{\speaksfont{Subtle}\speaksdel\hspace{\Dlabelsep}}
\setlength\speaksindent{0in}

\DramPer

\begin{drama*}

\scene

\StageDir{Enter \face, \subtle and \dol.}

\facespeaks Believe 't, I will.\\
\subtlespeaks \vinphantom{Believe 't, I will.} Thy worst. I fart at thee.\\
\dolspeaks Have you your wits? why, gentlemen! for love--\\
\facespeaks Sirrah, I'll strip you--\\
\subtlespeaks \vinphantom{Sirrah, I'll strip you--} What to do? Lick figs\\
  Out at my--\\
\facespeaks \vinphantom{Out at my--} Rogue, rogue!--out of all your sleights.\\
\dolspeaks Nay, look ye, sovereign, general, are you madmen?\\
\subtlespeaks O, let the wild sheep loose. I'll gum your silks\\
  With good strong water, an you come.\\
\dolspeaks \vinphantom{With good strong water, an you come.} Will you have\\
  The neighbours hear you? will you betray all?\\
  Hark! I hear somebody.\\
\facespeaks \vinphantom{Hark! I hear somebody.} Sirrah--\\
\subtlespeaks \vinphantom{Hark! I hear somebody. Sirrah--} I shall mar\\
  All that the tailor has made, if you approach.\\
\facespeaks You most notorious whelp, you insolent slave,\\
  Dare you do this?\\
\subtlespeaks \vinphantom{Dare you do this?} Yes, faith; yes, faith.\\
\facespeaks \vinphantom{Dare you do this? Yes, faith; yes, faith.} Why, who\\
  Am I, my mungrel? who am I?\\
\subtlespeaks \vinphantom{Am I, my mungrel? who am I?} I'll tell you.,\\
  Since you know not yourself.\\
\facespeaks \vinphantom{Since you know not yourself.} Speak lower, rogue.\\
\subtlespeaks Yes, you were once (time's not long past) the good,\\
  Honest, plain, livery-three-pound-thrum, that kept\\
  Your master's worship's house here in the Friars,\\
  For the vacations--\\
\facespeaks \vinphantom{For the vacations--} Will you be so loud?

\end{drama*}
